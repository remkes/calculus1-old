\begin{align*} \frac{d}{dx} x^n \amp = \lim_{h \rightarrow 0} \frac{(x+h)^n - x^n}{h} =\\ \amp = \lim_{h \rightarrow 0} \frac{x^n + nx^{n-1}h + \binom{n}{2} x^{n-2}h^2 + \ldots + nxh^{n-1} + h^n - x^n}{h}\\ \amp = \lim_{h \rightarrow 0} \frac{nx^{n-1}h + \binom{n}{2} x^{n-2}h^2 + \ldots + nxh^{n-1} + h^n}{h}\\ \amp = \lim_{h \rightarrow 0} nx^{n-1} + \binom{n}{2} x^{n-2}h + \ldots + nxh^{n-2} + h^{n-1}\\ \amp = nx^{n-1} \end{align*}
