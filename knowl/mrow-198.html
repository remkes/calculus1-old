\begin{align*} \frac{d^3}{dx^3} e^{4x} \amp = \frac{d}{dx} \frac{d}{dx} \frac{d}{du} e^u \Big|_{u = 4x} \frac{d}{dx} 4x\\ \amp = \frac{d}{dx} \frac{d}{dx} e^u \Big|_{u = 4x} 4\\ \amp = 4 \frac{d}{dx} \frac{d}{dx} e^{4x} \\ \amp = 4 \frac{d}{dx} \frac{d}{du} e^{u} \Big|_{u=4x} \frac{d}{dx} 4x \\ \amp = 4 \frac{d}{dx} e^{u} \Big|_{u=4x} 4 \\ \amp = 16 \frac{d}{dx} e^{4x}\\ \amp = 16 \frac{d}{du} e^{u} \Big|_{u=4x} \frac{d}{dx} 4x \\ \amp = 16 e^{u} \Big|_{u=4x} 4 = 64 e^{4x} \end{align*}
