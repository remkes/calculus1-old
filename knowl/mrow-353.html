\begin{align*} \frac{dy}{dx} \amp = \frac{\sin x}{y}\\ y dy \amp = \sin x dx\\ \frac{y^2}{2} \amp = - \cos x + c\\ y \amp = \pm \sqrt{c - 2\cos x}\\ \text{ Initial condition: } y(0) \amp = 1\\ 1 \amp = \sqrt{c - 2 \cos (0)} = \sqrt{c-2}\\ 1 \amp = c-2\\ c \amp = 3\\ y \amp = \sqrt{3 - 2 \cos x} \end{align*}
